319 template<typename T>
320 void CircularBuffer<T>::enqueue(const T &data, tm_page & new_tm_page) {
321 
322  uint32_t tempCounter;
323  size_t tempIndex;
324  timeabs tmptime; // verify type ... was uint64_t --> change to timeabs
325  timespan nextTime;
326  uint32_t newtempCounter;
327  Event<T> *tempEvent;
328  tm_page * ntp;
329 
330  // this part need to be processed atomically (set timestamp and increment the counter)
331  // since several less significant bytes are available in one 64 bit word, we will use
332  // it to store the counter. the counter should be larger as much as the buffer size.
333  FRAME_ADDRESS_subtype new_value = 0; // this value may be used with incorrect initialization [DO NOT MAKE CHANGES BELOW]
334 
335  // instructions for measure time and number of tries
336  //START_MEASURE();
337 
338  ATOMIC_begin_VALUE64(getFrameReference());
339 
340  //COUNT_CYCLE();
341 
342  // lets increment and get the counter atomically
343  tempCounter = getCounterValue(OLD_FRAME_ADDRESS);
344  // map counter to buffer index
345  tempIndex = counterToIndex(tempCounter);
346 
347  // get timestamp
348  tmptime = clockgettime();
349 
350  DEBUGV3("Time:%lld - %lld = %lld : Pointer: %p _> %p\n",
351  tmptime,
352  getCounterCurrentTimestamp(OLD_FRAME_ADDRESS),
353  tmptime - getCounterCurrentTimestamp(OLD_FRAME_ADDRESS),
354  new_tm_page,
355  &local_tm_page
356  );
357 
358  nextTime = tmptime - getCounterCurrentTimestamp(OLD_FRAME_ADDRESS) ;
359 
360  // case new_tm_page is currently in use then swap it with the local_tm_page
361  if ( getCounterCurrentPage(OLD_FRAME_ADDRESS) == (uint32_t)&new_tm_page )
362  {
363  DEBUGV3("using main page\n");
364  ntp = &local_tm_page;
365  }
366  else
367  {
368  ntp = &new_tm_page;
369  }
370 
371  ntp->current_time = tmptime;
372 
373  // lets create the new value using the old value with incremented counter and new page
374  newtempCounter = tempCounter + 1;
375 
376  setCounterCurrentPage(new_value, (uint32_t)ntp);
377  setCounterValue(new_value, newtempCounter);
378 
379  // lets mark the event updatable
380  ca_accesspointer[tempIndex].state.store(UPDATABLE);
381 
382  DEBUGV3("Time:%lld -- %ld : Pointer: %p\n",
383  ntp->current_time,
384  nextTime,
385  ntp
386  );
387 
388  ATOMIC_end_VALUE64(new_value, getFrameReference());
389 
390  // now we know the new index and timestamp
391  // lets modify the new event and set it to ready state
392 
393  tempEvent = &ca_accesspointer[tempIndex].ev;
394  tempEvent->setTime( (timespan) nextTime );
395  tempEvent->setData( data );
396 
397  ca_accesspointer[tempIndex].state.store(READY);
398 
399  //STOP_MEASURE();
400 
401  DEBUGV3("Counter is lock free:%d\n", atomic_is_lock_free(&getFrameReference()));
402 }

411 void CircularBuffer<T>::getState(timeabs &time, size_t &idx) const
412 {
413  /* Frame can be swapped when time and idx are retrieved; a guarantee for
414  * frame swapping is ensured by comparing the frame address before and
415  * after the assignments.

418  time = getCounterCurrentTimestamp(OLD_FRAME_ADDRESS);
